测试器

测试器的编写主要是完成主测试器和子测试器的编写，子测试器的编写涉及到Tester和Performance的框架的实现，其中Tester主要进行测试电路的搭建，而Performance主要是实现对仿真结果的处理。

Tester

使用Tester框架实现子测试器时，有以下三个要素必须实现：

  • 继承基类TesterBase。
  • testbench搭建。
  • 父类方法performance重写。

具体结构模板如下所示：

class Tester(TesterBase)
    def __init__(self: module, module: "Module", name: str):
        ...
    def testbench(self):
        """
        测试电路搭建，使用self.dut()来获取生成器的wrapper()
        """
        ...
    def performance(self):
        return CustomPerformance(self)
方法说明
init()类初始化，module和name参数一定要有。可以加入其他的参数
testbench()搭建仿真测试平台并进行仿真。
performance()返回自定义的Performance(CustomPerformance)。这里返回的实际上是CustomPerformance中performance方法所返回的字典

Performance

使用Performance框架实现仿真结果处理时，有以下三个要素必须实现：

  • 继承基类Performance。
  • 父类方法performance重写。
  • 对仿真结果进行处理。

具体结构模板如以下所示：

class CustomPerformance(Performance):
    def performance(self):
        return {"key1": self.get_perforamnce_name_1(),
                "key2": self.get_perforamnce_name_2(),
                ...
                "keyN": self.get_perforamnce_name_N()
                }

    def get_perforamnce_name_1(self):
        ...
        return result

    def get_perforamnce_name_2(self):
        ...
        return result
    ...

    def get_perforamnce_name_N(self):
        ...
        return result
方法说明
performance()父类performance方法重写，返回存储处理结果的字典。
get_performance_name_N()仿真结果处理方法，方法命名自拟

运放生成器--OTA5T

class Single_Stage_PMOS_Input_OTA5T(OpampGenerator):
    def __init__(self):
        super().__init__("diso")
        lmin = self.lmin
        u = 1e-6

        self.l1 = 0.5 * u
        self.w1 = 16 * u
        self.m1 = 14

        self.l3 = 0.9 * u
        self.w3 = 8 * u
        self.m3 = 1

        self.l0 = 0.5 * u
        self.w0 = 5 * u
        self.m0 = 6

        self.l7 = self.l0
        self.w7 = self.w0
        self.m7 = 1

    def core(self, VINP, VINN, IDC: Ctrl, Vo: Output):

        vtail, vcur = Node(2)
        l1, w1, m1 = self.l1, self.w1, self.m1
        l0, w0, m0 = self.l0, self.w0, self.m0
        l3, w3, m3 = self.l3, self.w3, self.m3
        l7, w7, m7 = self.l7, self.w7, self.m7

        Vdd() >> (pm7 := PMOS(l=l7, w=w7, m=m7) % (IDC)) >> IDC
        Vdd() >> (pm0 := PMOS(l=l0, w=w0, m=m0) % (IDC)) >> vtail
        vcur >> (nm3 := NMOS(l=l3, w=w3, m=m3) % (vcur)) >> Vss()
        Vo >> (nm4 := NMOS(l=l3, w=w3, m=m3) % (vcur)) >> Vss()
        vcur >> (nm1 := PMOS(l=l1, w=w1, m=m1)) % (VINP) >> vtail
        Vo >> (nm2 := PMOS(l=l1, w=w1, m=m1)) % (VINN) >> vtail

    def wrapper(self, VINP, VINN, Vo: Output):

        IDC = Node()
        dut = self.build()
        (VINP, VINN) >> dut % IDC >> Vo
        (IDC, Vss()) >> ISource(dc=10e-6)
        C(Vo, Vss(), c=5e-12).rename("load")

class OTATester(TesterBase):
    def __init__(self, module, **kwargs):
        _setenv, kwargs = get_setenv(kwargs)
        with setenv(**_setenv):
            self.init(module, **kwargs)

    def init(
        self,
        module,
        name,
        vdd,
        vcmi=None,
        vcmo=None,
        OTAtype="diso",
        mode=2,
        params={},
        skip_pz: bool = True,
        figpath=None,
        tbname="normal",
        save_op=False,
        save_currents=False,
        actester_settings={},
        srtester_settings={},
        noisetester_settings={},
        thdtester_settings={},
        **kwargs,
    ):
        self.name = name
        self.module = module
        self.vdd = vdd
        self.vcmi = vcmi
        self.vcmo = vcmo
        self.OTAtype = OTAtype
        self.mode = mode
        self.save_op = save_op
        self.save_currents = save_currents
        self.actester_settings = actester_settings
        self.srtester_settings = srtester_settings
        self.noisetester_settings = noisetester_settings
        self.thdtester_settings = thdtester_settings
        self.corner = get_corners()[0]
        self.temperature = getenv("temp")

        figpath = (
            f"{figpath}"
            if figpath
            else f"{get_ted_data()}/modules/circuits/{name}/{tbname}/"
        )

        self.figpath = rf"{figpath}/waveform/"
        self.mc_figpath = rf"{figpath}/monte_carlo_fig/"

        self.mc_table = None
        self.sim_table = None
        self.mc_fig_lib = []
        self.waveform_lib = []
        self.perf = {}

        stage = getenv("stage")
        if skip_pz is not None:
            self.skip_pz: bool = skip_pz
        else:
            self.skip_pz: bool = get_simulator() != "spectre" or stage == "post"
        self.build()

    def build(self):
        module, name, vdd, OTAtype, vcmi, vcmo = (
            self.module,
            self.name,
            self.vdd,
            self.OTAtype,
            self.vcmi,
            self.vcmo,
        )

        get_circuit().parallel.start() #开启并行

        self.gainTester = GainTester(module, name, vdd, OTAtype, vcmi, vcmo, self.actester_settings).run()
        self.THDTester = THDTester(module, name, vdd, OTAtype, vcmi, vcmo, self.thdtester_settings).run()
        self.DCTester = DCTester(module, name, vdd, OTAtype, vcmi, vcmo).run()
        self.psrrTester = PSRRTester(module, name, vdd, OTAtype, vcmi, vcmo).run()
        self.noiseTester = NoiseTester(module, name, vdd, OTAtype, vcmi, vcmo, self.noisetester_settings).run()
        self.cmrrTester = CMRRTester(module, name, vdd, OTAtype, vcmi, vcmo).run()
        self.srTester = SRTester(module, name, vdd, OTAtype, vcmi, vcmo, self.srtester_settings).run()
        self.ICMRTester = ICMRTestr(module, name, vdd, OTAtype, vcmi, vcmo).run()
        self.OPRTester = OutpSwingTester(module, name, vdd, OTAtype, vcmi, vcmo).run()

        get_circuit().parallel.stop() #结束并行
        

        self.perf["dc"] = self.DCTester
        self.perf["gain"] = self.gainTester
        self.perf["psrr"] = self.psrrTester
        self.perf["cmrr"] = self.cmrrTester
        self.perf["noise"] = self.noiseTester
        self.perf["sr"] = self.srTester
        self.perf["icmr"] = self.ICMRTester
        self.perf["ops"] = self.OPRTester

        "从子测试器中获得仿真处理结果，这里的键为Performance框架中performance()返回的字典的键"
        self.thd = self.THDTester["thd"]
        self.gain = self.gainTester["gain"]
        self.power = self.DCTester["power"]
        self.BW = self.gainTester["bw"]
        self.GBW = self.gainTester["gbw"]
        self.UGB = self.gainTester["ugb"]
        self.ICMRmin_OPE = self.ICMRTester["ICMRmin_OPE"]
        self.ICMRmax_OPE = self.ICMRTester["ICMRmax_OPE"]
        self.ICMRerro = self.ICMRTester["ICMRerro"]
        self.OPSmin_OPE = self.OPRTester["OPSmin_OPE"]
        self.OPSmax_OPE = self.OPRTester["OPSmax_OPE"]
        self.Phase_margin = self.gainTester["pm"]
        self.SRp, self.SRn = self.srTester["SRp"], self.srTester["SRn"]
        self.thermal_noise = self.noiseTester["thermal"]
        self.f1_noise = self.noiseTester["f1"]
        self.noise_1M = self.noiseTester["noise_1M"]
        self.CMRR = self.cmrrTester["cmrr"]
        self.PSRRp = self.psrrTester["psrr"]
        self.PSRR_1K = self.psrrTester["psrr@1K"]
        self.PSRR_1M = self.psrrTester["psrr@1M"]
        self.offset = 0

        self.summary()
        if get_simulator() not in ["spectremdl", "alpsmeas"]:
            self.plot()

    def plot(self):
        from pyted import rmdir

        rmdir(self.figpath)

        self.gainTester.plot(figpath=self.figpath, tester=self)
        self.psrrTester.plot(figpath=self.figpath, tester=self)
        self.cmrrTester.plot(figpath=self.figpath, tester=self)
        self.noiseTester.plot(figpath=self.figpath, tester=self)
        self.srTester.plot(figpath=self.figpath, tester=self)

    def plot_mc(self):
        from pyted import rmdir

        rmdir(self.mc_figpath)

        summarys = [
            self.DCTester.mc_summary(),
            self.gainTester.mc_summary(),
            self.psrrTester.mc_summary(),
            self.cmrrTester.mc_summary(),
            self.noiseTester.mc_summary(),
            self.srTester.mc_summary(),
        ]
        for i in summarys:
            for k, v in i.items():
                if v:
                    v.plot(
                        name=k,
                        ckt_name=self.name,
                        figpath=self.mc_figpath,
                        tester=self,
                    )

    def monte_calro_summary(self):
        summarys = [

            self.gainTester.mc_summary(),
            self.psrrTester.mc_summary(),
            self.cmrrTester.mc_summary(),

#demo1 以OTATester为例

otatb = OTAtester(module,ckt_name,OTAtype,vdd,vcmi)
'获取性能。以从gainTester中获取gain性能为例'
otatb.gainTester["gain"]

根据以上模板，在主测试台中获取某个子测试台的某个具体性能语法为： MainTestbench.SubTestench["performance_name"]
